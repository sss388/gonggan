# Vue + Nest + MySQL
## frontend
- yarn create vuetify
- yarn add -D tailwindcss postcss autoprefixer
- yarn tailwindcss init -p
- yarn add vuex@next --save
- yarn add axios
- yarn add @heroicons/vue@latest

## backend
- yarn add global @nestjs/cli
- nest new backend
- yarn add -D prisma
- yarn add @prisma/client
- yarn prisma init
- nest g mo prisma
- nest g s prisma --no-spec
- yarn add @nestjs/config
- yarn add --save class-validator class-transformer
- yarn add @nestjs/jwt passport-jwt @nestjs/passport
- yarn add argon2
